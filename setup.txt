# Base: i3 window manager, Xorg, btop system monitor
apt-get install i3 xorg btop

# media keys / OSD (see README "Media keys" section)
#   wpctl ships with wireplumber (pulled in by PipeWire); dunst is the notification
#   daemon — themed Catppuccin by apply-theme.sh (layout template + palettes
#   vendored in ~/setup/dunst; see README "Notifications (dunst)"). Needs
#   dunst >= 1.7 for the dunstrc.d/ drop-in (this machine has 1.13).
apt-get install brightnessctl playerctl libnotify-bin dunst

# wallpaper (see README "Wallpaper") — feh --bg-fill
apt-get install feh

# lock screen: blurred-wallpaper i3lock (see README "Lock screen").
#   i3lock = the locker; imagemagick pre-renders the blurred background.
#   Without imagemagick, lock.sh still locks (solid Catppuccin base colour).
apt-get install i3lock imagemagick

# transparency / compositor (see README "Transparency (compositor)")
apt-get install picom

# external monitors: hotplug profiles + rotation + clamshell (see README
# "External monitors"). autorandr = EDID-matched display profiles (its packaged
# udev/systemd hook auto-applies them; lid events use libinput, already pulled in
# by xorg). arandr = optional drag-to-arrange GUI. After installing, run
#   ~/setup/displays/install.sh
# to (re)install the hotplug/lid fallbacks, the postswitch hook and rotate.sh.
apt-get install autorandr arandr

# status bar: polybar + rofi (rofi powers the $mod+d app launcher AND the
# polybar power menu) (see README "Status bar (polybar)" and "App launcher (rofi)")
apt-get install polybar rofi

# tray applets (dock into the polybar tray):
#   network-manager-gnome -> nm-applet (Wi-Fi)
#   blueman               -> blueman-applet / blueman-manager (Bluetooth)
apt-get install network-manager-gnome blueman

# audio mixer GUI (polybar volume right-click) + Bluetooth stack (bluetoothctl)
apt-get install pavucontrol bluez

# zsh + Catppuccin syntax highlighting (see README zsh) — plus oh-my-zsh (manual)
apt-get install zsh zsh-syntax-highlighting

# Brave/Chromium file dialogs: force the GTK portal backend so they pick up the
# GTK theme (see README "GTK apps"). Usually already installed.
apt-get install xdg-desktop-portal-gtk

# tmux + clipboard for copy-mode; git pulls in the plugins (see README "tmux").
# The config is themed Rosé Pine (NOT Catppuccin); apply-theme.sh seeds
# ~/.tmux.conf if absent and clones TPM + rose-pine into ~/.tmux/plugins.
apt-get install tmux xclip git

# Neovim Ruby LSP (ruby-lsp) build deps: native gems like `pg` compile against
# the PostgreSQL client headers (libpq-fe.h) with a C toolchain. Without these,
# ruby-lsp starts but its per-project bundle won't build in a Rails app (see
# README "Neovim"). Ruby itself comes from mise; ruby-lsp is installed by
# apply-theme.sh (`gem install ruby-lsp`). Neovim (>= 0.11) + git must be present
# so apply-theme.sh can bootstrap LazyVim and seed the vendored ~/setup/nvim config.
apt-get install libpq-dev build-essential

#   - Icons: Papirus -> git clone PapirusDevelopmentTeam/papirus-icon-theme
#     into ~/.local/share/icons, recolour folders with papirus-folders (no sudo)
#   - Neovim (>= 0.11): install separately (Debian stable's may be too old — use
#     unstable/backports or the appimage). Then apply-theme.sh BOOTSTRAPS LazyVim
#     (clones the starter to ~/.config/nvim if absent), SEEDS the vendored config
#     in ~/setup/nvim (keymaps/options/LSP/formatting; leader = "\"), installs the
#     global ruby-lsp gem, and pins the Catppuccin theme. Needs git on PATH.
#     See README "Neovim (LazyVim)".
#   - File manager: yazi (replaces ranger). Not in Debian repos — install the
#     prebuilt binary to ~/.local/bin (no sudo); opens files in nvim via $EDITOR
#     (set in ~/.zshrc). See README "File manager (yazi)" + cheatsheet
#     ~/setup/yazi-cheatsheet.md. Run `y` (cd-on-exit wrapper) or `yazi`.
#     Themed Catppuccin (mauve) by apply-theme.sh; sources vendored in ~/setup/yazi.
#       curl -fsSL -o /tmp/yazi.zip \
#         https://github.com/sxyazi/yazi/releases/latest/download/yazi-x86_64-unknown-linux-gnu.zip
#       unzip -j /tmp/yazi.zip '*/yazi' '*/ya' -d ~/.local/bin
#     Image preview in Alacritty needs Überzug++ (NOT the apt `ueberzug`); install
#     the .deb from the maintainer's OBS repo (needs root):
#       curl -fsSL -o /tmp/ueberzugpp.deb \
#         "https://download.opensuse.org/repositories/home:/justkidding/Debian_Unstable/amd64/ueberzugpp_2.9.10_amd64.deb"
#       sudo apt install /tmp/ueberzugpp.deb
#     Optional richer previews:  sudo apt install ffmpegthumbnailer poppler-utils imagemagick chafa
#   - Spotify: not in Debian repos — install spotify-client from Spotify's apt
#     repo (repository.spotify.com). Bound to $mod+s (opens on ws10 + plays);

#   - Mouse natural scroll: install the Xorg snippet (needs root) so the BT mouse

#            /etc/X11/xorg.conf.d/90-natural-scrolling.conf
